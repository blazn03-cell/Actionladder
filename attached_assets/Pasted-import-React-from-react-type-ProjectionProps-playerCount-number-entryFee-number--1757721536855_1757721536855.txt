import React from "react";

type ProjectionProps = {
  playerCount: number;
  entryFee: number; // in cents
  subscriptionFee: number; // in cents
  matchesPerMonth: number;
};

const IncomeProjection: React.FC<ProjectionProps> = ({
  playerCount,
  entryFee,
  subscriptionFee,
  matchesPerMonth,
}) => {
  const calculateProjectedIncome = () => {
    const entryTotal = playerCount * entryFee * matchesPerMonth;
    const subscriptionRevenue = playerCount * subscriptionFee;
    const totalProjectedIncome = entryTotal + subscriptionRevenue;

    return {
      entryTotal,
      subscriptionRevenue,
      totalProjectedIncome,
    };
  };

  const { entryTotal, subscriptionRevenue, totalProjectedIncome } =
    calculateProjectedIncome();

  return (
    <div className="p-4 rounded-lg bg-gray-800 border border-gray-600 mb-4">
      <h4 className="text-lg font-semibold text-white mb-2">
        Projection for {playerCount} Players
      </h4>
      <p className="text-gray-300">
        Projected Entry Fees: ${(entryTotal / 100).toLocaleString()}
      </p>
      <p className="text-gray-300">
        Projected Subscription Revenue: $
        {(subscriptionRevenue / 100).toLocaleString()}
      </p>
      <p className="text-xl font-bold text-green-400">
        Total Projected Income: ${(totalProjectedIncome / 100).toLocaleString()}
      </p>
    </div>
  );
};

// Usage inside MonetizationDashboard
const MonetizationDashboard = () => {
  const scenarios = [
    { playerCount: 10, entryFee: 10000, subscriptionFee: 7900, matchesPerMonth: 2 },
    { playerCount: 20, entryFee: 10000, subscriptionFee: 7900, matchesPerMonth: 2 },
    { playerCount: 30, entryFee: 10000, subscriptionFee: 7900, matchesPerMonth: 4 },
  ];

  return (
    <div>
      {/* Other components */}

      <h3 className="text-2xl font-bold text-white mb-4">
        Future Income Projections
      </h3>
      {scenarios.map((s, i) => (
        <IncomeProjection key={i} {...s} />
      ))}

      {/* Other components */}
    </div>
  );
};

export default MonetizationDashboard;
