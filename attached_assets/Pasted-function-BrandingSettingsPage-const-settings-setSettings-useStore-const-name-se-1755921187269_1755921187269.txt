function BrandingSettingsPage(){
  const { settings, setSettings } = useStore();
  const [name, setName]   = React.useState(settings.hallName || "Action Ladder");
  const [slug, setSlug]   = React.useState(settings.hallSlug || "action");
  const [logo, setLogo]   = React.useState(settings.hallLogo || "");
  const [primary, setPrimary] = React.useState(settings.primaryColor || "#0ea5e9");
  const [qr, setQr]       = React.useState(settings.cashAppQRUrl || "");
  function save(){
    setSettings({ ...settings, hallName:name, hallSlug:slug, hallLogo:logo, primaryColor:primary, cashAppQRUrl:qr });
    alert("Branding saved.");
  }
  return (
    <div className="space-y-3 text-sm">
      <h2 className="text-2xl font-bold">ðŸŽ¨ Branding</h2>
      <div className="grid md:grid-cols-2 gap-3">
        <label className="block"><div className="text-xs font-medium">Hall name</div><input className="w-full border rounded-xl px-3 py-2" value={name} onChange={(e)=>setName(e.target.value)} /></label>
        <label className="block"><div className="text-xs font-medium">Slug</div><input className="w-full border rounded-xl px-3 py-2" value={slug} onChange={(e)=>setSlug(e.target.value)} /></label>
        <label className="block"><div className="text-xs font-medium">Logo URL</div><input className="w-full border rounded-xl px-3 py-2" value={logo} onChange={(e)=>setLogo(e.target.value)} /></label>
        <label className="block"><div className="text-xs font-medium">Primary color</div><input type="color" className="w-full border rounded-xl px-3 py-2" value={primary} onChange={(e)=>setPrimary(e.target.value)} /></label>
        <label className="block md:col-span-2"><div className="text-xs font-medium">Cash App QR (optional)</div><input className="w-full border rounded-xl px-3 py-2" value={qr} onChange={(e)=>setQr(e.target.value)} placeholder="https://.../your-qr.png" /></label>
      </div>
      <button className="px-3 py-2 rounded-xl border" onClick={save}>Save Branding</button>
      {logo && <div className="mt-2"><img alt="logo preview" src={logo} className="h-12" /></div>}
    </div>
  );
}
